    let mut out = String::new();

    out.push_str(&emit_schemas(document));

**2c. `emit_plans(document)` — one entry per `WorkflowDecl`:**

For each `workflow Foo(x: string) -> T { ... }`, emit:

```javascript
const PLANS = {
  Foo: {
    name: "Foo",
    requiredArgs: ["x"],
    returnType: "T",   // null if no return type
    steps: [ /* compiled step array */ ],
  },
  // ...
};
```

Step compilation rules (v1 — per Spec 38 §3.2):

| Statement kind | Compiled step type |
| --- | --- |
| `Statement::LetDecl` with `Expr::ExecuteRun { .. }` as value | `{ type: "execute_agent", bind: name, agent: X, taskTemplate: "...", requireType: T }` |
| `Statement::ExecuteRun { .. }` (no binding) | `{ type: "execute_agent", bind: "_", agent: X, taskTemplate: "...", requireType: T }` |
| `Statement::Return { .. }` | `{ type: "return", value: { type: "var", name: X } }` or `{ type: "return", value: { type: "literal", value: V } }` |
| `Statement::ForLoop`, `Statement::IfCond` (v1) | Single `{ type: "execute_agent", bind: "_", agent: <first_agent_in_body>, taskTemplate: "<full loop/condition as task context>", requireType: null }` — emit W-RUN01 at build time |

Template string interpolation: in `taskTemplate`, replace `${varName}` with `${scope.varName}` for runtime interpolation. The template is stored as a JS template literal string.

**2h. `emit_workflow_executor()` — static, returns the same JS for every document:**

```javascript
function interpolate(template, scope) {
  return template.replace(/\$\{(\w+)\}/g, (_, k) => scope[k] ?? "");
}

function parseAndValidate(output, typeName) {
  let parsed;
  try {
    parsed = typeof output === "string" ? JSON.parse(output) : output;
  } catch {
    throw { code: "E-RUN03", message: `LLM output is not valid JSON for type "${typeName}"` };
  }
  const schema = SCHEMAS[typeName];
  if (!schema) return parsed;  // no schema = no validation
  validateAgainstSchema(parsed, schema, typeName);
  return parsed;
}

function validateAgainstSchema(value, schema, typeName) {
  if (schema.type !== "object") return;
  for (const req of (schema.required ?? [])) {
    if (!(req in value)) {
      throw {
        code: "E-RUN03",
        message: `type validation failed — return type "${typeName}" not satisfied: missing field "${req}"`,
      };
    }
  }
}

function resolveValue(valueSpec, scope) {
  if (valueSpec.type === "var")     return scope[valueSpec.name];
  if (valueSpec.type === "literal") return valueSpec.value;
  return null;
}

async function executeWorkflow(planName, args) {
  const plan = PLANS[planName];
  if (!plan) {
    const available = Object.keys(PLANS).join(", ");
    throw { code: "E-RUN01", message: `workflow "${planName}" not found\n  available: ${available}\n  hint: run \`claw build\` to recompile if you added a new workflow` };
  }

  for (const required of plan.requiredArgs) {
    if (!(required in args)) {
      throw { code: "E-RUN02", message: `missing required argument "${required}" for workflow "${planName}"\n  usage: claw run ${planName} --arg ${required}=<value>` };
    }
  }

  const scope = { ...args };

  for (const step of plan.steps) {
    if (step.type === "execute_agent") {
      const task = interpolate(step.taskTemplate, scope);
      const output = await runAgent(step.agent, task);
      scope[step.bind] = step.requireType
        ? parseAndValidate(output, step.requireType)
        : output;
    } else if (step.type === "return") {
      return resolveValue(step.value, scope);
    }
  }
  return null;
}
```

function parseArgs(rawArgs) {
  const result = {};
  for (let i = 0; i < rawArgs.length; i++) {
    if (rawArgs[i] === "--arg" && i + 1 < rawArgs.length) {
      const eq = rawArgs[++i].indexOf("=");

    case "E-RUN01": return 1;
    case "E-RUN02": return 2;
    case "E-RUN03": return 3;
    case "E-RUN04": return 4;
    case "E-RUN05": return 5;

}

```rust
fn run_run(args: RunArgs) -> Result<(), ClawCliError> {
    // Walk up from CWD to find nearest claw.json
    let project_root = find_project_root().ok_or_else(|| {
        ClawCliError::Message("no claw.json found in current directory or any parent".to_owned())
    })?;

    let runtime_path = project_root.join("generated").join("runtime.js");
    if !runtime_path.exists() {
        eprintln!(
            "error[E-RUN05]: runtime not built — generated/runtime.js not found\n  run `claw build` first"
        );
        std::process::exit(5);
    }

    let node = find_node_binary();

    let mut cmd = Command::new(&node);
    cmd.arg(&runtime_path);
    cmd.arg(&args.workflow);
    for kv in &args.args {
        cmd.arg("--arg");
        cmd.arg(kv);
    }
    if let Some(client) = &args.client {
        cmd.arg("--client");
        cmd.arg(client);
    }

    // Set timeout via separate thread if timeout > 0
    let mut child = cmd.spawn().map_err(|e| {
        ClawCliError::Message(format!("failed to spawn node: {e}"))
    })?;

    let exit_status = if args.timeout > 0 {
        let timeout = Duration::from_secs(args.timeout);
        let pid = child.id();
        let handle = std::thread::spawn(move || {
            std::thread::sleep(timeout);
            // Send SIGTERM on timeout
            #[cfg(unix)]
            unsafe { libc::kill(pid as i32, libc::SIGTERM); }
        });
        let status = child.wait().map_err(|e| ClawCliError::Message(e.to_string()))?;
        drop(handle);
        status
    } else {
        child.wait().map_err(|e| ClawCliError::Message(e.to_string()))?
    };

    let code = exit_status.code().unwrap_or(1);

    // Write output to file if --output specified
    if let Some(out_path) = &args.output {
        // Capture stdout instead of piping to ours
        // Re-implement with output() if --output is set
        // Simplification: for now, just exit with the node process code
        // Full --output impl is Task 4
        let _ = out_path;
    }

    std::process::exit(code);
}
```

## Invariants — never violate these

1. **All tests pass after every task group.** Run `INSTA_UPDATE=always ~/.cargo/bin/cargo test` after each.
2. **`runtime.js` is always generated for `--lang opencode`.** No flag or condition should skip it. It is a peer of `mcp-server.js`.
3. **`runtime.js` agent runners call LLM API directly.** Never emit `execSync`, `exec`, `spawnSync`, or any subprocess call to `opencode` or `opencode -p`. This is the same invariant as `mcp-server.js`.
4. **`shared_js.rs` is the single source of LLM loop JS.** After Task 1, `emit_llm_loop_anthropic` and `emit_llm_loop_ollama` must exist in exactly one place.
5. **No new DSL syntax, no parser changes, no AST changes.** This spec is purely codegen + CLI. If you find yourself editing `src/ast.rs` or `src/parser.rs`, stop.
6. **`claw run` never auto-builds.** If `generated/runtime.js` is missing, exit with code 5 and tell the user to run `claw build`. No silent LLM calls during `claw run`.
7. **Exit codes from `node generated/runtime.js` are passed through unchanged.** The Rust wrapper must not remap or swallow them.
8. **`claw serve` stub is acceptable for now.** Full HTTP implementation is not gated by this spec — the stub that prints a message and exits 0 is correct for this iteration.
9. **All `Statement` match arms must include `Statement::Reason { .. }`.** Add this arm to every `match statement` you write in `emit_plans`. It should emit a `execute_agent` step for now (same as complex control flow).
10. **W-RUN01 is emitted for `for`/`if` blocks** during plan compilation (build time, not runtime). Use `eprintln!` with the format `warning[W-RUN01]: ...` — same style as existing compiler warnings.
